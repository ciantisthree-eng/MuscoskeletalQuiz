// Set up the global variables (MANDATORY)

const firebaseConfig = typeof __firebase_config !== 'undefined' ? JSON.parse(__firebase_config) : {};

const initialAuthToken = typeof __initial_auth_token !== 'undefined' ? __initial_auth_token : undefined;

window.auth = null;

window.firebaseApp = initializeApp(firebaseConfig);

window.db = getFirestore(window.firebaseApp);

window.auth = getAuth(window.firebaseApp);

// Authentication logic

async function initializeAuth() {

if (!window.auth) return;

try {

if (initialAuthToken) {

await signInWithCustomToken(window.auth, initialAuthToken);

} else {

await signInAnonymously(window.auth);

}

// Optional: Store initial score data (not used in this simple quiz but included for completeness)

const userId = window.auth.currentUser?.uid || 'anonymous';

const scoreDocRef = doc(collection(window.db, `artifacts/${appId}/users/${userId}/quiz_scores`), 'current_session');

await setDoc(scoreDocRef, { initialized: new Date().toISOString() }, { merge: true });

 

} catch (error) {

console.error("Firebase authentication failed:", error);

}

}

 

if (window.auth) {

initializeAuth();

}

window.handleOptionClick = function(event, option) {

if (userSelectedOption !== null) return; // Prevent double clicking

 

userSelectedOption = option;

const isCorrect = (option === quizQuestions[currentQuestionIndex].answer);

 

if (isCorrect) {

score++;

event.target.classList.add('correct');

} else {

event.target.classList.add('incorrect');

}

 

// Highlight the correct answer if the user was wrong

if (!isCorrect) {

const correctButton = Array.from(document.querySelectorAll('.option-button')).find(btn => btn.dataset.option === quizQuestions[currentQuestionIndex].answer);

if (correctButton) {

correctButton.classList.add('correct');

}

}

 

renderQuiz(); // Re-render to show feedback and Next button

}

 

window.handleNextClick = function() {

currentQuestionIndex++;

userSelectedOption = null;

renderQuiz();

} else {

quizState = 'finished';

}

// --- API Call for Question Generation ---

 

async function generateRemainingQuestions(existingQuestionsCount) {

const loadingStatus = document.getElementById('loading-status');

const progressBar = document.getElementById('progress-bar-loading');

 

const systemPrompt = "You are a quiz generation expert. Your task is to generate multiple-choice questions about the human muscular system based on general anatomy and physiology concepts, matching the complexity and topics (muscle types, structure, contraction, specific muscle names, actions, and disorders) found in typical high school/college biology notes. Generate the remaining questions needed to reach a total of 100 questions in JSON format. Do not repeat the initial topics or questions, but ensure diversity. The topics covered are: Muscle Types, Structure of Muscle Fiber, Contraction Mechanism, Energy Source (Aerobic/Anaerobic), Neuromuscular Junction, Impairments (Tetanus, Dystrophy), Muscle Actions (Agonist/Antagonist), Specific Muscles of the Face, Neck, Trunk, Shoulder, Arm, Forearm, Thigh, and Leg. Always provide exactly 4 options. The final JSON array must contain exactly 75 questions.";

 

const questionsToGenerate = QUIZ_LENGTH - existingQuestionsCount;

 

const userQuery = `I have ${existingQuestionsCount} questions already. Please generate exactly ${questionsToGenerate} more multiple-choice questions. Ensure the output is ONLY a single JSON array of objects, where each object has: 'question', 'options' (an array of 4 strings), and 'answer' (the correct option string).`;

 

loadingStatus.textContent = `Generating remaining questions (${existingQuestionsCount}/${QUIZ_LENGTH})...`;

progressBar.style.width = `${(existingQuestionsCount / QUIZ_LENGTH) * 100}%`;

 

 

const apiUrl = `https://generativelanguage.googleapis.com/v1beta/models/${modelName}:generateContent?key=${apiKey}`;

 

const payload = {

contents: [{ parts: [{ text: userQuery }] }],

systemInstruction: { parts: [{ text: systemPrompt }] },

generationConfig: {

responseMimeType: "application/json",

responseSchema: {

type: "ARRAY",

items: {

type: "OBJECT",

properties: {

"question": { "type": "STRING", "description": "The quiz question." },

"options": {

"type": "ARRAY",

"items": { "type": "STRING" },

"description": "Exactly four answer options."

},

"answer": { "type": "STRING", "description": "The correct answer, must match one of the options exactly." }

},

"required": ["question", "options", "answer"]

}

}

}

};

 

const maxRetries = 5;

let retryDelay = 1000;

 

for (let i = 0; i < maxRetries; i++) {

try {

const response = await fetch(apiUrl, {

method: 'POST',

headers: { 'Content-Type': 'application/json' },

body: JSON.stringify(payload)

});

 

if (!response.ok) {

throw new Error(`HTTP error! status: ${response.status}`);

}

 

const result = await response.json();

const candidate = result.candidates?.[0];

 

if (candidate && candidate.content?.parts?.[0]?.text) {

const jsonText = candidate.content.parts[0].text;

const generatedQuestions = JSON.parse(jsonText);

 

loadingStatus.textContent = `Generated ${generatedQuestions.length} questions. Finalizing quiz...`;

progressBar.style.width = `100%`;

 

// Filter to ensure correct structure

const validGeneratedQuestions = generatedQuestions.filter(q =>

q.question && Array.isArray(q.options) && q.options.length === 4 && q.answer

);

 

return validGeneratedQuestions;

} else {

throw new Error("API response structure invalid or content missing.");

}

} catch (error) {

console.error(`Attempt ${i + 1} failed:`, error);

if (i < maxRetries - 1) {

await new Promise(resolve => setTimeout(resolve, retryDelay));

retryDelay *= 2; // Exponential backoff

} else {

console.error("Failed to generate questions after all retries.");

return []; // Return empty array to use only initial questions

}

}

}

return [];

}

 

// --- Initialization ---

 

async function startQuiz() {

quizState = 'loading';

renderLoading();

 

// Reset state

quizQuestions = [];

currentQuestionIndex = 0;

score = 0;

userSelectedOption = null;

 

// 1. Load initial questions

quizQuestions.push(...initialQuestions);

 

// 2. Generate remaining questions

const generated = await generateRemainingQuestions(initialQuestions.length);

quizQuestions.push(...generated);

 

// 3. Ensure we have exactly 100 questions (or as close as possible)

quizQuestions = quizQuestions.slice(0, QUIZ_LENGTH);

 

// Shuffle the complete question array

for (let i = quizQuestions.length - 1; i > 0; i--) {

const j = Math.floor(Math.random() * (i + 1));

[quizQuestions[i], quizQuestions[j]] = [quizQuestions[j], quizQuestions[i]];

}

 

quizState = 'started';

renderQuiz();

}

window.handleNextClick = window.handleNextClick;

 

// Start the process when the window loads

window.onload = startQuiz;
